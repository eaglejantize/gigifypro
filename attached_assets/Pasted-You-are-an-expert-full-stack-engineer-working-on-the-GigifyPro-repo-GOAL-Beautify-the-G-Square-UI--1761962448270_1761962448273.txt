import { useEffect, useState } from "react";
export default function ThemeToggle(){
  const [dark,setDark] = useState(false);
  useEffect(()=>{ const saved = localStorage.getItem("gp_theme"); if(saved){ setDark(saved==="dark"); document.documentElement.classList.toggle("dark", saved==="dark"); }},[]);
  function toggle(){
    const next = !dark;
    setDark(next);
    localStorage.setItem("gp_theme", next?"dark":"light");
    document.documentElement.classList.toggle("dark", next);
  }
  return (
    <button onClick={toggle} className="gp-pill">{dark?"‚òÄÔ∏è Light":"üåô Dark"}</button>
  );
}
