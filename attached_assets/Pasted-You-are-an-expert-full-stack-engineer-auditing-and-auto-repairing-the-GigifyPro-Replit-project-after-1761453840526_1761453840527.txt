import { Router } from "express";
import { PrismaClient } from "@prisma/client";
const prisma = new PrismaClient();
const r = Router();

r.get("/debug/info", async (req,res)=>{
  const routes = (req.app as any)?._router?.stack
    ?.filter((l:any)=>l.route)
    ?.map((l:any)=>Object.keys(l.route.methods).map(m=>`${m.toUpperCase()} ${l.route.path}`))
    ?.flat() || [];
  res.json({
    env: {
      NODE_ENV: process.env.NODE_ENV || "dev",
      PARTNER_FEE_PERCENT: process.env.PARTNER_FEE_PERCENT || "(missing)",
      PLATFORM_FEE_PERCENT: process.env.PLATFORM_FEE_PERCENT || "(missing)",
      MIN_CHARGE_USD: process.env.MIN_CHARGE_USD || "(missing)",
      MIN_BLOCK_MINUTES: process.env.MIN_BLOCK_MINUTES || "(missing)"
    },
    routes
  });
});

r.get("/debug/counts", async (_req,res)=>{
  const p = new PrismaClient();
  const out:any = {};
  try {
    out.users = await p.user.count();
  } catch(e){}
  try { out.products = await p.product.count(); } catch(e){}
  try { out.services = (await import("../content/serviceInfo.json").catch(()=>({services:[]}))).services?.length || 0; } catch(e){ out.services = -1; }
  try { out.posts = await p.post.count(); } catch(e){}
  try { out.topics = await p.topic.count(); } catch(e){}
  try { out.events = await p.analyticsEvent.count(); } catch(e){}
  res.json(out);
});

export default r;
